133 inline void motorA(int16_t speed) {
134 bool clockwise = speed > 0;
135 speed = constrain(speed, -255, 255);
136 speed = abs(speed);
137 digitalWrite(ain1Pin, clockwise ? LOW : HIGH);
138 #if TWO_PINS_SINGLE_MOTOR
139 digitalWrite(ain2Pin, clockwise ? HIGH : LOW);
140 #endif
141 analogWrite(pwmaPin, speed);
142 digitalWrite(stbyPin, HIGH);
143 DEBUG_PRINT(DEBUG_MOTOR, "[speed: " + String(speed) + "] MotorA moving " + (clockwise ? "Clockwise" : "Counter-clockwise"));
144 }
