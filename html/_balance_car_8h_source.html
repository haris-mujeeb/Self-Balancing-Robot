80void balanceCar() {
81 sei(); // set enable interrupts
82 encoder_left_pulse_num_speed += pwm_left < 0 ? -encoder_count_left_a : encoder_count_left_a;
83 encoder_right_pulse_num_speed += pwm_right < 0 ? -encoder_count_right_a : encoder_count_right_a;
84 encoder_count_left_a = 0;
85 encoder_count_right_a = 0;
86 mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
87 kalmanfilter.Angle(ax, ay, az, gx, gy, gz, dt, Q_angle, Q_gyro, R_angle, C_0, K1);
88 kalmanfilter_angle = kalmanfilter.angle;
89 double balance_control_output = kp_balance * (kalmanfilter_angle - angle_zero) + kd_balance * (kalmanfilter.Gyro_x - angular_velocity_zero);
90
91 speed_control_period_count++;
92 if (speed_control_period_count >= 8) {
93 speed_control_period_count = 0;
94 double car_speed = (encoder_left_pulse_num_speed + encoder_right_pulse_num_speed) * 0.5;
95 encoder_left_pulse_num_speed = 0;
96 encoder_right_pulse_num_speed = 0;
97 encoder_speed_filtered = encoder_speed_filtered_old * 0.7 + car_speed * 0.3;
98 encoder_speed_filtered_old = encoder_speed_filtered;
99 car_speed_integeral += encoder_speed_filtered;
100 car_speed_integeral += -setting_car_speed;
101 car_speed_integeral = constrain(car_speed_integeral, -3000, 3000);
102 speed_control_output = -kp_speed * encoder_speed_filtered - ki_speed * car_speed_integeral;
103 rotation_control_output = setting_turn_speed + kd_turn * kalmanfilter.Gyro_z;
104 }
105
106 pwm_left = balance_control_output - speed_control_output - rotation_control_output;
107 pwm_right = balance_control_output - speed_control_output + rotation_control_output;
108 pwm_left = constrain(pwm_left, -255, 255);
109 pwm_right = constrain(pwm_right, -255, 255);
110
111 #if (DEBUG_CONTROL)
112 debugMsg += "[pwn_left:" + String(pwm_left) + "]";
113 debugMsg += "[pwn_right:" + String(pwm_right) + "]";
114 #endif
115
116 // Print Debug messages
117#ifdef DEBUG_PID_PITCH
118 debugMsg += "[PITCH PID: " + String(balance_control_output) + "]";
119#endif
120
121#ifdef DEBUG_PID_YAW
122 debugMsg += "[YAW PID: " + String(rotation_control_output) + "]";
123#endif
124
125#ifdef DEBUG_PID_POSITION
126 debugMsg += "[POS PID:" + String(speed_control_output) + "]";
127 debugMsg += "[postion:" + String(car_speed_integeral) + "]";
128#endif`
129
130#if (DEBUG_CONTROL) || (DEBUG_PID_PITCH) || (DEBUG_PID_YAW) || (DEBUG_PID_POSITION)
131 DEBUG_PRINT(DEBUG_MODE, debugMsg);
132 debugMsg = "";
133#endif
134
135#if (PLOT_MODE)
136 plotMsg += String(kalmanfilter_angle) + ",";
137 plotMsg += String(kalmanfilter.Gyro_z) + ",";
138 plotMsg += String(car_speed_integeral) + ",";
139 plotMsg += String(balance_control_output) + ",";
140 plotMsg += String(rotation_control_output) + ",";
141 plotMsg += String(speed_control_output) + ",";
142 plotMsg += String(pwm_left) + ",";
143 plotMsg += String(pwm_right);
144 SEND_FOR_PLOT(plotMsg);
145 plotMsg = "";
146#endif
147
148 if (motion_mode != START && motion_mode != STOP && (kalmanfilter_angle < balance_angle_min || balance_angle_max < kalmanfilter_angle))
149 {
150 motion_mode = STOP;
151 carStop();
152 }
153
154 if (motion_mode == STOP && key_flag != '4')
155 {
156 car_speed_integeral = 0;
157 setting_car_speed = 0;
158 pwm_left = 0;
159 pwm_right = 0;
160 carStop();
161 }
162 else if (motion_mode == STOP)
163 {
164 car_speed_integeral = 0;
165 setting_car_speed = 0;
166 pwm_left = 0;
167 pwm_right = 0;
168 }
169 else
170 {
171 if (pwm_left < 0)
172 {
173 digitalWrite(AIN1, 1);
174 analogWrite(PWMA_LEFT, -pwm_left);
175 }
176 else
177 {
178 digitalWrite(AIN1, 0);
179 analogWrite(PWMA_LEFT, pwm_left);
180 }
181 if (pwm_right < 0)
182 {
183 digitalWrite(BIN1, 1);
184 analogWrite(PWMB_RIGHT, -pwm_right);
185 }
186 else
187 {
188 digitalWrite(BIN1, 0);
189 analogWrite(PWMB_RIGHT, pwm_right);
190 }
191 }
192}
